# compute the sum of n numbers - without arrays
entry

entry_main()
    number n
    number sum -> 0

    read(n)

    entry_for i -> 1, n, 1:
        read(x)
        sum -> sum + x
    exit_for

    show(sum)

exit_main

exit

# compute the sum of n numbers - with arrays
entry

entry_main()
    number n
    number sum -> 0
    array->number numbers[]

    read(n)

    entry_for i -> 1, n, 1:
        read(x)
        numbers -> numbers + x
    exit_for

    entry_for i -> 1, n, 1:
        sum -> sum + numbers[i]
    exit_for

    show(sum)

exit_main

exit
